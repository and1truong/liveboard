# P1 — REST API Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the `/api/v1/*` REST API as an additive layer alongside the existing HTMX app, providing the stable contract the future iframe-renderer architecture will consume. No UI change; current app keeps working.

**Architecture:** All new routes mount under `/api/v1`. Existing `/api/*` (unversioned) routes stay intact. A single `POST /api/v1/boards/{slug}/mutations` endpoint dispatches all board mutations via a tagged-union `MutationOp` type, collapsing dozens of per-op endpoints into one. SSE is exposed at `/api/v1/events?board={slug}` backed by the existing broker in `internal/web/sse.go`. A `/api/versions` probe advertises supported REST versions.

**Tech Stack:** Go 1.24, chi/v5 router, existing `board.Engine`, existing `web.SSEBroker`. No new dependencies.

**Spec:** `docs/superpowers/specs/2026-04-15-iframe-renderer-architecture-design.md`
**Plan of plans:** `docs/superpowers/plans/2026-04-15-iframe-renderer-plan-of-plans.md`

**Out of scope for P1:**
- Authentication / API tokens (spec: "no auth for now")
- TypeScript parity (that's P2)
- Any change to `internal/web/` handlers or the HTMX frontend
- Deleting or modifying existing `/api/*` unversioned routes
- Renderer, shell, localStorage

**Conventions used in this plan:**
- Existing helpers (`respond`, `respondError`, `respondCreated`, `handleError`, `decodeJSON`, `pathParam`) in `internal/api/helpers.go` are reused — do not re-invent them.
- New files live in `internal/api/v1/` as a subpackage.
- Tests live next to source files as `*_test.go`, using the standard library's `testing` package and `net/http/httptest`.
- Every task ends with a commit. Use conventional commits: `feat(api): ...`, `test(api): ...`.

```go
func jsonContentType(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		w.Header().Set("Content-Type", "application/json")
		next.ServeHTTP(w, r)
```

```go
package v1

import (
	"encoding/json"
	"errors"
	"net/http"
	"os"

	"github.com/and1truong/liveboard/internal/board"
	"github.com/and1truong/liveboard/internal/workspace"
)

type errorResponse struct {
	Error  string `json:"error"`
	Code   string `json:"code"`
	Status int    `json:"status"`
}

// errInvalid is a generic "bad request" sentinel for v1-layer validation.
var errInvalid = errors.New("invalid request")

// writeError maps engine/workspace errors to HTTP status + canonical error code.
// The error code string is the protocol-level code used by the shell adapter
// (e.g. "VERSION_CONFLICT") — it must stay in sync with the spec.
func writeError(w http.ResponseWriter, err error) {
	status := http.StatusInternalServerError
	code := "INTERNAL"
	switch {
	case errors.Is(err, board.ErrNotFound), errors.Is(err, os.ErrNotExist):
		status, code = http.StatusNotFound, "NOT_FOUND"
	case errors.Is(err, board.ErrVersionConflict):
		status, code = http.StatusConflict, "VERSION_CONFLICT"
	case errors.Is(err, workspace.ErrAlreadyExists):
		status, code = http.StatusConflict, "ALREADY_EXISTS"
	case errors.Is(err, board.ErrOutOfRange):
		status, code = http.StatusBadRequest, "OUT_OF_RANGE"
	case errors.Is(err, workspace.ErrInvalidBoardName), errors.Is(err, errInvalid):
		status, code = http.StatusBadRequest, "INVALID"
	}
	w.WriteHeader(status)
	_ = json.NewEncoder(w).Encode(errorResponse{
		Error:  err.Error(),
		Code:   code,
		Status: status,
	})
}

func decodeJSON(r *http.Request, v any) error {
	return json.NewDecoder(r.Body).Decode(v)
}
```

## Task 6: Define `MutationOp` union type

```go
import (
	"github.com/and1truong/liveboard/internal/board"
)

// Dispatch executes a MutationOp against the engine.
// clientVersion is passed through to engine methods that support it;
// pass -1 to bypass the version check (for tests or idempotent reads).
//
// Engine methods that do not take a clientVersion currently perform their
// own version handling inside MutateBoard. For the REST API we prefer the
// version-aware variants where they exist.
func Dispatch(eng *board.Engine, boardPath string, clientVersion int, op MutationOp) error {
	switch op.Type {
	case "add_card":
		if op.AddCard == nil {
			return fmt.Errorf("add_card: missing params")
		}
		_, err := eng.AddCard(boardPath, op.AddCard.Column, op.AddCard.Title, op.AddCard.Prepend)
		return err
	case "move_card":
		return eng.MoveCard(boardPath, op.MoveCard.ColIdx, op.MoveCard.CardIdx, op.MoveCard.TargetColumn)
	case "reorder_card":
		return eng.ReorderCard(boardPath, op.ReorderCard.ColIdx, op.ReorderCard.CardIdx, op.ReorderCard.BeforeIdx, op.ReorderCard.TargetColumn)
	case "edit_card":
		p := op.EditCard
		return eng.EditCard(boardPath, p.ColIdx, p.CardIdx, p.Title, p.Body, p.Tags, p.Priority, p.Due, p.Assignee)
	case "delete_card":
		return eng.DeleteCard(boardPath, op.DeleteCard.ColIdx, op.DeleteCard.CardIdx)
	case "complete_card":
		return eng.CompleteCard(boardPath, op.CompleteCard.ColIdx, op.CompleteCard.CardIdx)
	case "tag_card":
		return eng.TagCard(boardPath, op.TagCard.ColIdx, op.TagCard.CardIdx, op.TagCard.Tags)
	case "add_column":
		return eng.AddColumn(boardPath, op.AddColumn.Name)
	case "rename_column":
		return eng.RenameColumn(boardPath, op.RenameColumn.OldName, op.RenameColumn.NewName)
	case "delete_column":
		return eng.DeleteColumn(boardPath, op.DeleteColumn.Name)
	case "move_column":
		return eng.MoveColumn(boardPath, op.MoveColumn.Name, op.MoveColumn.AfterCol)
	case "sort_column":
		return eng.SortColumn(boardPath, op.SortColumn.ColIdx, op.SortColumn.SortBy)
	case "toggle_column_collapse":
		return eng.ToggleColumnCollapse(boardPath, op.ToggleColumnCollapse.ColIdx)
	case "update_board_meta":
		p := op.UpdateBoardMeta
		return eng.UpdateBoardMeta(boardPath, p.Name, p.Description, p.Tags)
	case "update_board_members":
		return eng.UpdateBoardMembers(boardPath, op.UpdateBoardMembers.Members)
	case "update_board_icon":
		return eng.UpdateBoardIcon(boardPath, op.UpdateBoardIcon.Icon)
	case "update_board_settings":
		return eng.UpdateBoardSettings(boardPath, op.UpdateBoardSettings.Settings)
	default:
		return fmt.Errorf("unknown mutation op type: %q", op.Type)
	}
}
```

```go
type mutationRequest struct {

	Op            MutationOp `json:"op"`

func (d Deps) postMutation(w http.ResponseWriter, r *http.Request) {
	slug := chi.URLParam(r, "slug")
	path, err := d.Workspace.SlugToPath(slug) // use whatever helper exists
	if err != nil {
		writeError(w, err)
		return
	}

	var req mutationRequest
	if err := decodeJSON(r, &req); err != nil {
		writeError(w, fmt.Errorf("%w: %v", errInvalid, err))
		return
	}

	if err := Dispatch(d.Engine, path, req.ClientVersion, req.Op); err != nil {
		writeError(w, err)
		return
	}

	// Return the updated board + publish SSE.
	board, err := d.Workspace.LoadBoard(slug)
	if err != nil {
		writeError(w, err)
		return
	}
	d.SSE.Publish(slug)
	_ = json.NewEncoder(w).Encode(board)
}
```

```go
func (d Deps) getBoardSettings(w http.ResponseWriter, r *http.Request) {
	slug := chi.URLParam(r, "slug")
	board, err := d.Workspace.LoadBoard(slug)
	if err != nil {
		writeError(w, err)
		return
	}
	global := web.LoadSettingsFromDir(d.Workspace.Dir)
	resolved := web.ResolveSettings(global, board.Settings)
	_ = json.NewEncoder(w).Encode(resolved)
}

func (d Deps) putBoardSettings(w http.ResponseWriter, r *http.Request) {
	slug := chi.URLParam(r, "slug")
	path, err := d.Workspace.SlugToPath(slug)
	if err != nil {
		writeError(w, err)
		return
	}
	var patch models.BoardSettings
	if err := decodeJSON(r, &patch); err != nil {
		writeError(w, err)
		return
	}
	if err := d.Engine.UpdateBoardSettings(path, patch); err != nil {
		writeError(w, err)
		return
	}
	d.SSE.Publish(slug)
	w.WriteHeader(http.StatusNoContent)
}
```

```go
package v1

import (
	"fmt"
	"net/http"
)

func (d Deps) getEvents(w http.ResponseWriter, r *http.Request) {
	slug := r.URL.Query().Get("board")
	if slug == "" {
		writeError(w, fmt.Errorf("%w: board query param required", errInvalid))
		return
	}

	flusher, ok := w.(http.Flusher)
	if !ok {
		http.Error(w, "streaming not supported", http.StatusInternalServerError)
		return
	}

	w.Header().Set("Content-Type", "text/event-stream")
	w.Header().Set("Cache-Control", "no-cache")
	w.Header().Set("Connection", "keep-alive")
	w.Header().Set("X-Accel-Buffering", "no")

	ch := d.SSE.Subscribe(slug)
	defer d.SSE.Unsubscribe(slug, ch)

	_, _ = fmt.Fprintf(w, "event: connected\ndata: {}\n\n")
	flusher.Flush()

	for {
		select {
		case <-r.Context().Done():
			return
		case _, ok := <-ch:
			if !ok {
				return
			}
			// Load current version for the payload.
			version := 0
			if board, err := d.Workspace.LoadBoard(slug); err == nil {
				version = board.Version
			}
			_, _ = fmt.Fprintf(w, "event: board.updated\ndata: {\"board_id\":%q,\"version\":%d}\n\n", slug, version)
			flusher.Flush()
		}
	}
}
```

Create `scripts/smoke-api-v1.sh`:

```bash
#!/usr/bin/env bash
# Smoke test for /api/v1. Usage: BASE=http://localhost:7070 bash scripts/smoke-api-v1.sh
set -euo pipefail

BASE="${BASE:-http://localhost:7070}"
BOARD="${BOARD:-demo}"

pass() { printf "  \033[32m✓\033[0m %s\n" "$1"; }
fail() { printf "  \033[31m✗\033[0m %s\n" "$1"; exit 1; }

say() { printf "\n\033[1m%s\033[0m\n" "$1"; }

say "Versions probe"
curl -fsS "$BASE/api/versions" | grep -q '"current":"v1"' && pass "/api/versions" || fail "/api/versions"

say "Workspace"
curl -fsS "$BASE/api/v1/workspace" | grep -q '"dir"' && pass "/api/v1/workspace" || fail "/api/v1/workspace"

say "Boards list"
curl -fsS "$BASE/api/v1/boards" | grep -q '\[' && pass "/api/v1/boards" || fail "/api/v1/boards"

say "Board detail"
curl -fsS "$BASE/api/v1/boards/$BOARD" | grep -q '"columns"' && pass "/api/v1/boards/$BOARD" || fail "/api/v1/boards/$BOARD"

say "Board settings"
curl -fsS "$BASE/api/v1/boards/$BOARD/settings" > /dev/null && pass "GET settings" || fail "GET settings"

say "Mutation: add_card"
RESP=$(curl -fsS -X POST "$BASE/api/v1/boards/$BOARD/mutations" \
  -H 'Content-Type: application/json' \
  -d '{"client_version":-1,"op":{"type":"add_card","column":"Todo","title":"smoke-test"}}')
echo "$RESP" | grep -q '"version"' && pass "POST mutation" || fail "POST mutation"

say "Version conflict"
STATUS=$(curl -s -o /dev/null -w '%{http_code}' -X POST "$BASE/api/v1/boards/$BOARD/mutations" \
  -H 'Content-Type: application/json' \
  -d '{"client_version":0,"op":{"type":"add_card","column":"Todo","title":"stale"}}')
[ "$STATUS" = "409" ] && pass "409 on stale version" || fail "expected 409, got $STATUS"

say "SSE connects"
timeout 2 curl -fsSN "$BASE/api/v1/events?board=$BOARD" | head -n 1 | grep -q 'event:' && pass "SSE connect" || fail "SSE connect"

say "All smoke tests passed."
```
