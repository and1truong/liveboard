# P5.0 — JSON API v1 (Go) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Close the remaining gaps in the existing `internal/api/v1` package so it fully mirrors the `BackendAdapter` contract: add board create/rename/delete handlers, switch the SSE endpoint to a workspace-wide multiplex emitting `board.list.updated`, extend the SSE broker with workspace-level publish/subscribe, and add `Workspace.RenameBoard`.

**Architecture:** New routes (`POST /boards`, `PATCH /boards/{slug}`, `DELETE /boards/{slug}`) are thin shims over `workspace.Workspace`. After mutation, handlers call a new `SSEBroker.PublishBoardList()`. The existing per-board `GET /events?board=…` is replaced by a workspace-wide stream that fans in per-board events (`board.updated`) AND the new list-level event.

**Tech Stack:** Go 1.24, chi, existing `board.Engine`, `workspace.Workspace`, `web.SSEBroker`.

**Spec:** `docs/superpowers/specs/2026-04-15-p5-0-json-api-v1-design.md`

// RenameBoard moves <oldName>.md → <slugify(newName)>.md and updates the board's Name.
// Returns the resulting summary. ErrNotFound if old missing; ErrAlreadyExists if new collides; ErrInvalidBoardName if new name unusable.
func (w *Workspace) RenameBoard(oldName, newName string) (parser.BoardSummaryInfo, error) {
    newSlug := slugify(newName)
    if newSlug == "" {
        return parser.BoardSummaryInfo{}, fmt.Errorf("%w: empty after slugify", workspace.ErrInvalidBoardName)
    }
    oldPath, err := w.BoardPath(oldName)
    if err != nil {
        return parser.BoardSummaryInfo{}, err
    }
    newPath := filepath.Join(w.dir, newSlug+".md")
    if oldPath != newPath {
        if _, err := os.Stat(newPath); err == nil {
            return parser.BoardSummaryInfo{}, fmt.Errorf("%w: %s", workspace.ErrAlreadyExists, newSlug)
        }
    }
    b, err := w.LoadBoard(oldName)
    if err != nil { return parser.BoardSummaryInfo{}, err }
    b.Name = strings.TrimSpace(newName)
    b.Version++
    raw, err := writer.Render(b)
    if err != nil { return parser.BoardSummaryInfo{}, err }
    if oldPath != newPath {
        if err := os.WriteFile(newPath, raw, 0o644); err != nil {
            return parser.BoardSummaryInfo{}, err
        }
        if err := os.Remove(oldPath); err != nil {
            // best-effort: leave new in place; surface the error.
            return parser.BoardSummaryInfo{}, err
        }
    } else {
        if err := os.WriteFile(oldPath, raw, 0o644); err != nil {
            return parser.BoardSummaryInfo{}, err
        }
    }
    return parser.BoardSummaryInfo{Slug: newSlug, Name: b.Name, Icon: b.Icon, Version: b.Version}, nil
}
```

If the existing helper is `slugify(name)` not yet imported, follow the convention used by `CreateBoard`. If `writer` / `parser` packages are aliased differently, match the existing import block. If `BoardSummaryInfo` doesn't have an `Icon` or `Version` field, drop the corresponding lines.

// SubscribeBoardList registers a channel for workspace-list events (board.list.updated).
func (b *SSEBroker) SubscribeBoardList() (chan SSEEvent, func()) {
    b.mu.Lock()
    defer b.mu.Unlock()
    ch := make(chan SSEEvent, 8)
    b.boardList[ch] = struct{}{}
    return ch, func() {
        b.mu.Lock()
        defer b.mu.Unlock()
        if _, ok := b.boardList[ch]; ok {
            delete(b.boardList, ch)
            close(ch)
        }
    }
}

// PublishBoardList fans out a board.list.updated event to all workspace-list subscribers.
func (b *SSEBroker) PublishBoardList() {
    b.mu.RLock()
    defer b.mu.RUnlock()
    for ch := range b.boardList {
        select {
        case ch <- SSEEvent{Type: "board.list.updated"}:
        default:
        }
    }
}

// SubscribeAllBoards registers a channel that receives every board.updated event.
func (b *SSEBroker) SubscribeAllBoards() (chan SSEEvent, func()) {
    b.mu.Lock()
    defer b.mu.Unlock()
    ch := make(chan SSEEvent, 32)
    b.allBoards[ch] = struct{}{}
    return ch, func() {
        b.mu.Lock()
        defer b.mu.Unlock()
        if _, ok := b.allBoards[ch]; ok {
            delete(b.allBoards, ch)
            close(ch)
        }
    }
}
```

package v1

import (
    "encoding/json"
    "fmt"
    "net/http"
    "time"
)

func (d Deps) getEvents(w http.ResponseWriter, r *http.Request) {
    flusher, ok := w.(http.Flusher)
    if !ok {
        http.Error(w, "streaming not supported", http.StatusInternalServerError)
        return
    }
    w.Header().Set("Content-Type", "text/event-stream")
    w.Header().Set("Cache-Control", "no-cache")
    w.Header().Set("Connection", "keep-alive")
    w.Header().Set("X-Accel-Buffering", "no")

    listCh, cancelList := d.SSE.SubscribeBoardList()
    defer cancelList()
    boardCh, cancelBoards := d.SSE.SubscribeAllBoards()
    defer cancelBoards()

    fmt.Fprintf(w, "retry: 5000\n\n")
    flusher.Flush()

    ticker := time.NewTicker(15 * time.Second)
    defer ticker.Stop()

    for {
        select {
        case <-r.Context().Done():
            return
        case <-ticker.C:
            fmt.Fprintf(w, ": ping\n\n")
            flusher.Flush()
        case ev := <-listCh:
            fmt.Fprintf(w, "event: %s\ndata: null\n\n", ev.Type)
            flusher.Flush()
        case ev := <-boardCh:
            // ev.Payload is the slug; load to pull the current version.
            slug := ev.Payload
            version := 0
            if b, err := d.Workspace.LoadBoard(slug); err == nil {
                version = b.Version
            }
            payload, _ := json.Marshal(struct {
                BoardID string `json:"board_id"`
                Version int    `json:"version"`
            }{BoardID: slug, Version: version})
            fmt.Fprintf(w, "event: %s\ndata: %s\n\n", ev.Type, payload)
            flusher.Flush()
        }
    }
}
